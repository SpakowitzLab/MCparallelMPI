                      I=1
                      StopLoop=0
                      do while (StopLoop.EQ.0)
                         if (INDBIN.EQ.INDPHI(I).AND.I.LE.NPHI) then
                            StopLoop=1
                            if (isA) then
                               DPHIA(I)=DPHIA(I)+rrdr*WTOT*V !/Vol(INDBIN)
                            else
                               DPHIB(I)=DPHIB(I)+rrdr*WTOT*V !/Vol(INDBIN)
                            endif
                         elseif (I.GT.NPHI) then
                            StopLoop=1
                            NPHI=NPHI+1
                            INDPHI(I)=INDBIN

                            if (isA) then
                               DPHIA(I)=DPHIA(I)+rrdr*WTOT*V !/Vol(INDBIN)
                            else
                               DPHIB(I)=DPHIB(I)+rrdr*WTOT*V !/Vol(INDBIN)
                            endif
                         else
                            I=I+1
                         endif                     
                      enddo
